class Platform
{

#ifndef SCRIPTING_API
  Platform(::Platform* platform);

#endif

  /** Move platform until at given node, then stop */

  /** Start moving platform */

  /** Stop platform at next node */

  ::Platform* platform;

  Platform(const Platform&);

#endif
};
