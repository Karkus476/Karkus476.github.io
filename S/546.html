#ifndef HEADER_SUPERTUX_GUI_MENU_ACTION_HPP
#define HEADER_SUPERTUX_GUI_MENU_ACTION_HPP

/* Kinds of menu items */
enum MenuItemKind {
  MN_ACTION,
  MN_GOTO,
  MN_TOGGLE,
  MN_BACK,
  MN_INACTIVE,
  MN_TEXTFIELD,
  MN_NUMFIELD,
  MN_INTFIELD,
  MN_CONTROLFIELD,
  MN_STRINGSELECT,
  MN_BADGUYSELECT,
  MN_LABEL,
  MN_COLOR,
  MN_SCRIPT,
  MN_FILE,
  MN_REMOVE,
  MN_HL /* horizontal line */
};

/* Action done on the menu */
enum MenuAction {
  MENU_ACTION_NONE = -1,
  MENU_ACTION_UP,
  MENU_ACTION_DOWN,
  MENU_ACTION_LEFT,
  MENU_ACTION_RIGHT,
  MENU_ACTION_HIT,
  MENU_ACTION_INPUT,
  MENU_ACTION_REMOVE,
  MENU_ACTION_BACK
};

#endif // HEADER_SUPERTUX_GUI_MENU_ACTION_HPP
