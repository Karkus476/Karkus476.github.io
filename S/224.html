#include "control/codecontroller.hpp"

CodeController::CodeController()
{}

CodeController::~CodeController()
{}

void
CodeController::press(Control c, bool pressed_)
{
  controls[c] = pressed_;
}

void
CodeController::update()
{
  Controller::update();

  for(int i = 0; i < CONTROLCOUNT; ++i)
    controls[i] = false;
}
